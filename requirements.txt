# If you don't have pip installed, install it first
python -m ensurepip --upgrade

# Install Streamlit
pip install streamlit

# Install the OpenAI package
pip install openai

# Install YouTube Transcript API
pip install youtube-transcript-api

# Install Python dotenv for environment variable management
pip install python-dotenv

# Install LangChain and related packages for text splitting and embeddings
pip install langchain
pip install faiss-cpu

# Optionally, you can install the development version of LangChain if you need the latest features
# pip install git+https://github.com/hwchase17/langchain.git

# If you need additional callback functionality with OpenAI
pip install langchain-callbacks
